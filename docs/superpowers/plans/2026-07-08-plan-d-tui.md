# queohoh Plan D — Ink TUI Cockpit Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** The minimal walking-skeleton cockpit: an Ink app that sits in tmux tab 0 forever — live queue (wide left column) with full queue management (quick-add, run definition, retry/skip, answer needs-input, detail drill-in), worktrees + cron-slot right column, daemon auto-reconnect.

**Architecture:** `packages/tui` is a thin client: `ApiClient` connect + `subscribe` push → React state → Ink render. Pure formatting/selection helpers are separated from components for cheap unit tests; components tested with ink-testing-library against fixture snapshots; one integration smoke wires a real `ApiServer`+`Engine` (fake claude) to the real hook. Detail view reads report/transcript straight from the runs dir (files are truth). Guardrail from the spec: ~1s summary ticks, one drill-in at a time, no firehose rendering.

**Tech Stack:** ink 6, react 19, ink-testing-library, existing `@queohoh/daemon` (ApiClient, StateSnapshot) + `@queohoh/core` types. A tiny in-repo TextInput (no ink-text-input dep — avoids peer roulette).

**Spec:** `docs/superpowers/specs/2026-07-08-queohoh-slice1-design.md` (TUI dashboard section) · **Builds on:** Plan B socket API + Plan C notes (use `state`+`subscribe`, never the MCP projection; treat 5s call timeouts on discovery runDefinition as "verify by state").

## Global Constraints

- Node >= 22, TS strict ESM, `jsx: "react-jsx"`. Lint via `mise x node@22 -- pnpm lint`. No Co-Authored-By trailers.
- Two-column layout: QUEUE left (flexGrow 2, hosts detail drill-in), right column stacks CRON slot (placeholder box: "cron — slice 2") above WORKTREES.
- Status glyphs: running `▶`, queued `○`, needs-input `?`, done `✓`, failed `✗`.
- Keybindings (footer always visible): `↑/↓ or j/k` select · `a` add adhoc · `t` run task definition · `r` retry · `s` skip · `w` answer needs-input · `enter` detail · `esc` back/cancel · `q` quit.
- Queue shows: live tasks (all) + recent archived (last 10). Running rows show elapsed (1s tick). Queued rows show lane position. Failed rows show error.
- Daemon down → yellow banner "daemon unreachable — retrying…", auto-reconnect every 2s, TUI never crashes.
- All mutations go through the socket API (retry/skip/setWorktree/enqueue/runDefinition). Call failures render in a transient status line, never throw.
- Detail view: run report (report.md) + last 25 lines of transcript.md, refreshed every 1s while open; missing files → "(no report yet)". Files read from `runsPath(statePath())/<task-id>/`.
- Components take data via props; daemon I/O isolated in the `useDaemon` hook and `actions` module.

`packages/tui/src/format.ts`:

```ts
import type { TaskInstance, TaskStatus } from "@queohoh/core";
import type { StateSnapshot } from "@queohoh/daemon";

const GLYPHS: Record<TaskStatus, string> = {
	running: "▶",
	queued: "○",
	"needs-input": "?",
	done: "✓",
	failed: "✗",
};

export function statusGlyph(status: TaskStatus): string {
	return GLYPHS[status];
}

export function elapsed(sinceIso: string, now: number): string {
	const totalSec = Math.max(0, Math.floor((now - Date.parse(sinceIso)) / 1000));
	const hours = Math.floor(totalSec / 3600);
	const minutes = Math.floor((totalSec % 3600) / 60);
	const seconds = totalSec % 60;
	if (hours > 0) return `${hours}h${String(minutes).padStart(2, "0")}m`;
	if (minutes > 0) return `${minutes}m${String(seconds).padStart(2, "0")}s`;
	return `${seconds}s`;
}

export function promptSummary(prompt: string, width: number): string {
	const line =
		prompt.split("\n").find((l) => l.trim().length > 0)?.trim() ?? "";
	if (line.length <= width) return line;
	return `${line.slice(0, width - 1)}…`;
}

export interface QueueRow {
	id: string;
	glyph: string;
	lane: string;
	summary: string;
	detail: string;
	kind: "live" | "archived";
}

function laneLabel(task: TaskInstance): string {
	return `${task.target.repo}:${task.target.worktree ?? task.target.ref}`;
}

export function buildQueueRows(
	snapshot: StateSnapshot,
	now: number,
	width: number,
): QueueRow[] {
	const queuedPosition = new Map<string, number>();

	const liveRows = snapshot.tasks.map((task): QueueRow => {
		let detail: string;
		switch (task.status) {
			case "running":
				detail = `⏱ ${elapsed(task.created, now)}`;
				break;
			case "queued": {
				const lane = laneLabel(task);
				const position = (queuedPosition.get(lane) ?? 0) + 1;
				queuedPosition.set(lane, position);
				detail = `#${position} in lane`;
				break;
			}
			case "needs-input":
			case "failed":
				detail = task.error ?? task.status;
				break;
			case "done":
				detail = "done";
				break;
		}
		return {
			id: task.id,
			glyph: statusGlyph(task.status),
			lane: laneLabel(task),
			summary: promptSummary(task.prompt, width),
			detail,
			kind: "live",
		};
	});

	const archivedRows = snapshot.archivedRecent.slice(-10).map(
		(task): QueueRow => ({
			id: task.id,
			glyph: statusGlyph(task.status),
			lane: laneLabel(task),
			summary: promptSummary(task.prompt, width),
			detail: "archived",
			kind: "archived",
		}),
	);

	return [...liveRows, ...archivedRows];
}
```

`packages/tui/src/use-daemon.ts`:

```ts
import { useEffect, useRef, useState } from "react";
import type { StateSnapshot } from "@queohoh/daemon";
import { ApiClient } from "@queohoh/daemon";

export interface DaemonState {
	snapshot: StateSnapshot | null;
	connected: boolean;
}

export function useDaemon(
	sockPath: string,
	opts?: { retryMs?: number },
): DaemonState {
	const retryMs = opts?.retryMs ?? 2000;
	const [state, setState] = useState<DaemonState>({
		snapshot: null,
		connected: false,
	});
	const alive = useRef(true);

	useEffect(() => {
		alive.current = true;
		let client: ApiClient | null = null;
		let retryTimer: NodeJS.Timeout | null = null;

		const attempt = async () => {
			if (!alive.current) return;
			client = new ApiClient();
			try {
				await client.connect(sockPath);
				await client.subscribe((pushed) => {
					if (alive.current) {
						setState({ snapshot: pushed as StateSnapshot, connected: true });
					}
				});
				const initial = (await client.call("state")) as StateSnapshot;
				if (alive.current) setState({ snapshot: initial, connected: true });
				client.onClose(() => {
					if (!alive.current) return;
					setState((prev) => ({ ...prev, connected: false }));
					scheduleRetry();
				});
			} catch {
				client.close();
				if (alive.current) {
					setState((prev) => ({ ...prev, connected: false }));
					scheduleRetry();
				}
			}
		};

		const scheduleRetry = () => {
			if (retryTimer) clearTimeout(retryTimer);
			retryTimer = setTimeout(() => void attempt(), retryMs);
		};

		void attempt();
		return () => {
			alive.current = false;
			if (retryTimer) clearTimeout(retryTimer);
			client?.close();
		};
	}, [sockPath, retryMs]);

	return state;
}
```

`packages/tui/src/actions.ts`:

```ts
import { ApiClient } from "@queohoh/daemon";

export interface DefinitionSummary {
	repo: string;
	name: string;
	args: string[];
	hasDiscovery: boolean;
}

export interface Actions {
	enqueue(prompt: string, repo: string): Promise<string | null>;
	retry(id: string): Promise<string | null>;
	skip(id: string): Promise<string | null>;
	setWorktree(id: string, worktree: string): Promise<string | null>;
	runDefinition(
		repo: string,
		name: string,
		args: string[],
	): Promise<string | null>;
	definitions(): Promise<DefinitionSummary[]>;
}

async function withClient<T>(
	sockPath: string,
	fn: (client: ApiClient) => Promise<T>,
): Promise<T> {
	const client = new ApiClient();
	try {
		await client.connect(sockPath);
		return await fn(client);
	} finally {
		client.close();
	}
}

function asError(err: unknown): string {
	return err instanceof Error ? err.message : String(err);
}

export function createActions(sockPath: string): Actions {
	const mutate = async (
		method: string,
		params: Record<string, unknown>,
	): Promise<string | null> => {
		try {
			await withClient(sockPath, (c) => c.call(method, params));
			return null;
		} catch (err) {
			return asError(err);
		}
	};

	return {
		enqueue: (prompt, repo) => mutate("enqueue", { prompt, repo }),
		retry: (id) => mutate("retry", { id }),
		skip: (id) => mutate("skip", { id }),
		setWorktree: (id, worktree) => mutate("setWorktree", { id, worktree }),
		runDefinition: async (repo, name, args) => {
			const result = await mutate("runDefinition", { repo, name, args, source: "tui" });
			// discovery can exceed the client timeout — the tasks may still land;
			// the push subscription re-syncs, so treat timeout as success.
			if (result?.includes("timed out")) return null;
			return result;
		},
		definitions: async () => {
			try {
				return await withClient(
					sockPath,
					(c) => c.call("definitions") as Promise<DefinitionSummary[]>,
				);
			} catch {
				return [];
			}
		},
	};
}
```

```tsx
import type { StateSnapshot } from "@queohoh/daemon";
import { Box, Text } from "ink";
import React from "react";

export function RightColumn({ snapshot }: { snapshot: StateSnapshot }) {
	const lanes = new Map<string, "busy" | "paused" | "free">();
	for (const task of snapshot.tasks) {
		if (task.target.worktree === null) continue;
		const lane = `${task.target.repo}:${task.target.worktree}`;
		if (task.status === "running") lanes.set(lane, "busy");
		else if (task.status === "failed" && lanes.get(lane) !== "busy") {
			lanes.set(lane, "paused");
		} else if (!lanes.has(lane)) lanes.set(lane, "free");
	}
	const interactive = snapshot.sessions.filter((s) => s.kind === "interactive");

	return (
		<Box flexDirection="column" flexGrow={1}>
			<Box borderStyle="round" flexDirection="column" paddingX={1}>
				<Text bold>CRON</Text>
				<Text dimColor>cron — slice 2</Text>
			</Box>
			<Box borderStyle="round" flexDirection="column" paddingX={1}>
				<Text bold>WORKTREES</Text>
				{[...lanes.entries()].map(([lane, state]) => (
					<Text key={lane}>
						{lane} <Text dimColor>{state}</Text>
					</Text>
				))}
				{interactive.map((s) => (
					<Text key={s.key}>
						{s.cwd?.split("/").pop() ?? s.key} <Text color="yellow">YOU</Text>
					</Text>
				))}
				{lanes.size === 0 && interactive.length === 0 ? (
					<Text dimColor>no active worktrees</Text>
				) : null}
			</Box>
		</Box>
	);
}
```

```tsx
import { Text, useInput } from "ink";
import React from "react";

export function TextInput({
	label,
	value,
	onChange,
	onSubmit,
	onCancel,
}: {
	label: string;
	value: string;
	onChange: (value: string) => void;
	onSubmit: (value: string) => void;
	onCancel: () => void;
}) {
	useInput((input, key) => {
		if (key.return) {
			onSubmit(value);
		} else if (key.escape) {
			onCancel();
		} else if (key.backspace || key.delete) {
			onChange(value.slice(0, -1));
		} else if (input && !key.ctrl && !key.meta) {
			onChange(value + input);
		}
	});
	return (
		<Text>
			{label}&gt; {value}█
		</Text>
	);
}
```

```ts
import { existsSync, readFileSync } from "node:fs";
import { join } from "node:path";

export function readRunFiles(
	runsDir: string,
	taskId: string,
): { report: string | null; transcriptTail: string[] } {
	const dir = join(runsDir, taskId);
	const reportPath = join(dir, "report.md");
	const transcriptPath = join(dir, "transcript.md");
	const report = existsSync(reportPath)
		? readFileSync(reportPath, "utf-8")
		: null;
	const transcriptTail = existsSync(transcriptPath)
		? readFileSync(transcriptPath, "utf-8").split("\n").slice(-25)
		: [];
	return { report, transcriptTail };
}
```

```tsx
import { Box, Text, useInput } from "ink";
import React, { useEffect, useState } from "react";
import { readRunFiles } from "../run-files.js";

export function DetailView({
	taskId,
	runsDir,
	onBack,
}: {
	taskId: string;
	runsDir: string;
	onBack: () => void;
}) {
	const [files, setFiles] = useState(() => readRunFiles(runsDir, taskId));
	useEffect(() => {
		const timer = setInterval(
			() => setFiles(readRunFiles(runsDir, taskId)),
			1000,
		);
		return () => clearInterval(timer);
	}, [runsDir, taskId]);
	useInput((_input, key) => {
		if (key.escape) onBack();
	});
	return (
		<Box borderStyle="round" flexDirection="column" paddingX={1} flexGrow={1}>
			<Text bold>TASK {taskId.slice(-6)}</Text>
			<Text bold dimColor>
				── transcript (tail) ──
			</Text>
			{files.transcriptTail.length === 0 ? (
				<Text dimColor>(no transcript yet)</Text>
			) : (
				files.transcriptTail.map((line, i) => (
					<Text key={`${i}-${line.slice(0, 8)}`}>{line}</Text>
				))
			)}
			<Text bold dimColor>
				── report ──
			</Text>
			<Text>{files.report ?? "(no report yet)"}</Text>
		</Box>
	);
}
```

`packages/tui/src/App.tsx`:

```tsx
import { Box, Text, useApp, useInput } from "ink";
import React, { useEffect, useMemo, useState } from "react";
import type { Actions, DefinitionSummary } from "./actions.js";
import { DetailView } from "./components/DetailView.js";
import { Footer } from "./components/Footer.js";
import { QueuePanel } from "./components/QueuePanel.js";
import { RightColumn } from "./components/RightColumn.js";
import { TextInput } from "./components/TextInput.js";
import { buildQueueRows } from "./format.js";
import { useDaemon } from "./use-daemon.js";

type Mode =
	| { kind: "list" }
	| { kind: "add-prompt" }
	| { kind: "add-repo"; prompt: string }
	| { kind: "worktree-input"; taskId: string }
	| { kind: "def-pick"; defs: DefinitionSummary[]; index: number }
	| { kind: "def-args"; def: DefinitionSummary }
	| { kind: "detail"; taskId: string };

export function App({
	sockPath,
	runsDir,
	actions,
}: {
	sockPath: string;
	runsDir: string;
	actions: Actions;
}) {
	const { exit } = useApp();
	const { snapshot, connected } = useDaemon(sockPath);
	const [now, setNow] = useState(Date.now());
	const [selected, setSelected] = useState(0);
	const [mode, setMode] = useState<Mode>({ kind: "list" });
	const [input, setInput] = useState("");
	const [statusLine, setStatusLine] = useState<string | null>(null);

	useEffect(() => {
		const timer = setInterval(() => setNow(Date.now()), 1000);
		return () => clearInterval(timer);
	}, []);

	const rows = useMemo(
		() =>
			snapshot
				? buildQueueRows(snapshot, now, 60)
				: [],
		[snapshot, now],
	);
	const clampedSelected = Math.min(selected, Math.max(0, rows.length - 1));

	const act = (result: Promise<string | null>) => {
		void result.then((err) => setStatusLine(err));
	};

	useInput((char, key) => {
		if (mode.kind !== "list") {
			if (mode.kind === "def-pick") {
				if (key.escape) setMode({ kind: "list" });
				else if (key.upArrow || char === "k") {
					setMode({ ...mode, index: Math.max(0, mode.index - 1) });
				} else if (key.downArrow || char === "j") {
					setMode({
						...mode,
						index: Math.min(mode.defs.length - 1, mode.index + 1),
					});
				} else if (key.return) {
					const def = mode.defs[mode.index];
					if (!def) return;
					if (def.args.length > 0) {
						setInput("");
						setMode({ kind: "def-args", def });
					} else {
						act(actions.runDefinition(def.repo, def.name, []));
						setMode({ kind: "list" });
					}
				}
			}
			return; // input modes handled by TextInput
		}
		setStatusLine(null);
		if (char === "q") exit();
		else if (key.upArrow || char === "k") {
			setSelected(Math.max(0, clampedSelected - 1));
		} else if (key.downArrow || char === "j") {
			setSelected(Math.min(rows.length - 1, clampedSelected + 1));
		} else if (char === "a") {
			setInput("");
			setMode({ kind: "add-prompt" });
		} else if (char === "t") {
			void actions.definitions().then((defs) => {
				if (defs.length === 0) setStatusLine("no task definitions found");
				else setMode({ kind: "def-pick", defs, index: 0 });
			});
		} else {
			const row = rows[clampedSelected];
			if (!row || row.kind === "archived") return;
			if (char === "r") act(actions.retry(row.id));
			else if (char === "s") act(actions.skip(row.id));
			else if (char === "w") {
				setInput("");
				setMode({ kind: "worktree-input", taskId: row.id });
			} else if (key.return) setMode({ kind: "detail", taskId: row.id });
		}
	});

	if (mode.kind === "detail") {
		return (
			<Box flexDirection="column">
				<DetailView
					taskId={mode.taskId}
					runsDir={runsDir}
					onBack={() => setMode({ kind: "list" })}
				/>
				<Footer />
			</Box>
		);
	}

	return (
		<Box flexDirection="column">
			{!connected ? (
				<Text backgroundColor="yellow" color="black">
					{" daemon unreachable — retrying… "}
				</Text>
			) : null}
			<Box>
				<QueuePanel
					rows={rows}
					selectedIndex={clampedSelected}
					statusLine={statusLine}
				/>
				{snapshot ? <RightColumn snapshot={snapshot} /> : null}
			</Box>
			{mode.kind === "add-prompt" ? (
				<TextInput
					label="prompt"
					value={input}
					onChange={setInput}
					onSubmit={(v) => {
						setMode({ kind: "add-repo", prompt: v });
						setInput("");
					}}
					onCancel={() => setMode({ kind: "list" })}
				/>
			) : null}
			{mode.kind === "add-repo" ? (
				<TextInput
					label="repo"
					value={input}
					onChange={setInput}
					onSubmit={(v) => {
						act(actions.enqueue(mode.prompt, v));
						setMode({ kind: "list" });
					}}
					onCancel={() => setMode({ kind: "list" })}
				/>
			) : null}
			{mode.kind === "worktree-input" ? (
				<TextInput
					label="worktree"
					value={input}
					onChange={setInput}
					onSubmit={(v) => {
						act(actions.setWorktree(mode.taskId, v));
						setMode({ kind: "list" });
					}}
					onCancel={() => setMode({ kind: "list" })}
				/>
			) : null}
			{mode.kind === "def-pick" ? (
				<Box flexDirection="column" borderStyle="round" paddingX={1}>
					<Text bold>run task definition</Text>
					{mode.defs.map((def, i) => (
						<Text key={`${def.repo}/${def.name}`} inverse={i === mode.index}>
							{def.repo}/{def.name}
							{def.args.length > 0 ? ` (${def.args.join(", ")})` : ""}
							{def.hasDiscovery ? " [discovery]" : ""}
						</Text>
					))}
				</Box>
			) : null}
			{mode.kind === "def-args" ? (
				<TextInput
					label={`${mode.def.name} args (${mode.def.args.join(" ")})`}
					value={input}
					onChange={setInput}
					onSubmit={(v) => {
						act(
							actions.runDefinition(
								mode.def.repo,
								mode.def.name,
								v.trim().length > 0 ? v.trim().split(/\s+/) : [],
							),
						);
						setMode({ kind: "list" });
					}}
					onCancel={() => setMode({ kind: "list" })}
				/>
			) : null}
			<Footer />
		</Box>
	);
}
```
